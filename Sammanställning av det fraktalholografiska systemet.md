# Sammanfattning ‚Äî syfte, verkan, mening


- **Skript 1 ‚Äî ‚Äúmedvetande.py‚Äù (S1-test):** Simulerar uppdateringen $I_{t+1}=\Psi!\big(I_t\oplus I_{\mathrm{L},t}\oplus I_{\mathrm{N},t}\oplus I_\ast\big)$ och stoppar n√§r stabilitet uppn√•s, dvs. $d(I_{t+1},I_t)\le\varepsilon$. Det opererar som ett praktiskt test av **Sats S1** (fixpunktens existens/unikhet under kontraktion).
- **Skript 2 ‚Äî ‚Äúmedvetande_mi.py‚Äù (S2-test):** Estimerar $\mathrm{MI}!\big(S_\lambda(I_{\mathrm{N}}),S_\lambda(I_\ast)\big)$ √∂ver flera $\lambda$ och kontrollerar villkoret $\inf_\lambda \mathrm{MI}&gt;\kappa$. Om sant ‚áí icke-lokal komponent b√§r last i $C$ enligt **Sats S2**.

**Varf√∂r detta fungerar:**


- **Kontraktion** s√§kras genom en vikt‚Äêskalig $\Psi$ med normbegr√§nsning s√• att $d(\Psi(I),\Psi(J))\le q,d(I,J)$ med $0&lt;q&lt;1$, vilket ger konvergens enligt Banach.
- **Holografi + skalning** modelleras via $H$ och $S_\lambda$ f√∂r att approximera fraktalholografiskt bevarande.
- **Informationssammanfogning** $\oplus$ v√§ljs monotont och avbildar till $[0,1]$ s√• att totalstat $I_{\mathrm T}=I\oplus I_{\mathrm L}\oplus I_{\mathrm N}\oplus I_\ast$ h√•ller sig i dom√§nen.


# Kvalitetskontroll ‚Äî hittade fel och fixar


1. **Fel:** Metodnamnet `__call.me__` (ej giltigt i Python) och anropet `self.__call.me__(I)` i f√∂rsta skriptet.
**Fix:** Inf√∂rde `_core(...)` och korrekt `__call__(...)`.
2. **Fel:** Metodnamn `nonlocal(...)` krockar med Python-nyckelordet `nonlocal`.
**Fix:** Bytt till `nonlocal_(...)` och uppdaterat alla anrop.
3. **Stabilitet:** Standardparametrarna kunde ge l√•ngsam/utebliven ‚ÄúSTABIL‚Äù status.
**Fix:** St√§llde $\Psi$ n√•got mer kontraktiv ($q=0.6$) och en rimlig tr√∂skel $\varepsilon=10^{-4}$; ger typiskt konvergens p√• $\approx 20\text{‚Äì}40$ steg.

    def nonlocal_(self, I: List[float]) -&gt; List[float]:
        m = sum(I) / len(I)
        return clip01([x + self.drift * (m - x) for x in I])

class Psi:
    # $\Psi(I)=\sigma(WI+b)$, $\|W\|\lesssim q$ f√∂r kontraktion
    def __init__(self, n: int, q: float = 0.6, seed: int = 42) -&gt; None:
        self.rng = random.Random(seed)
        self.W = [[(self.rng.uniform(-1.0, 1.0)) * 0.2 for _ in range(n)] for __ in range(n)]
        for i in range(n):
            self.W[i][i] = 0.5
        self._scale_W(q)
        self.b = [self.rng.uniform(-0.2, 0.2) for _ in range(n)]
    def _scale_W(self, q: float) -&gt; None:
        norm = sum(abs(x) for row in self.W for x in row) / len(self.W)
        factor = q / max(norm, 1e-6)
        for i in range(len(self.W)):
            for j in range(len(self.W)):
                self.W[i][j] *= factor
    def _core(self, I: List[float]) -&gt; List[float]:
        z = []
        for i in range(len(self.W)):
            s = sum(self.W[i][j] * I[j] for j in range(len(self.W)))
            z.append(sigmoid(s + self.b[i]))
        return z
    def __call__(self, I: List[float]) -&gt; List[float]:
        return self._core(I)

# ---------- Model ----------
class Model:
    def __init__(self, n: int = 8, lam: float = 0.5, seed: int = 137) -&gt; None:
        self.n = n
        self.S = Scale(lam)
        self.H = Holo()
        self.O = Observation(seed=seed)
        self.P = Psi(n=n, q=0.6, seed=seed)
        self.I_star = [0.6 if i % 2 == 0 else 0.4 for i in range(n)]  # $I_\ast$

    def step(self, I: List[float]) -&gt; tuple[list[float], dict]:
        I_L = self.O.local(I)
        I_N = self.O.nonlocal_(I)
        I_T = add_oplus(add_oplus(I, self.I_star), add_oplus(I_L, I_N))
        I_proj = self.H(self.S(I_T))
        I_next = self.P(I_proj)
        return I_next, {"I_L": I_L, "I_N": I_N, "I_star": self.I_star, "I_proj": I_proj}

# ---------- MI Estimator ----------
def mutual_information(xs: List[float], ys: List[float], bins: int = 16, eps: float = 1e-12) -&gt; tuple[float,float]:
    """
    Histogram-baserad $\mathrm{MI}$ i nats och bits mellan tv√• listor i $[0,1]$.
    """
    if len(xs) != len(ys):
        raise ValueError("xs and ys must have same length")
    n = len(xs)
    if n == 0:
        return 0.0, 0.0
    edges = [i / bins for i in range(bins + 1)]
    import bisect
    def bin_index(val: float) -&gt; int:
        if val &lt;= 0.0: return 0
        if val &gt;= 1.0: return bins - 1
        k = bisect.bisect_right(edges, val) - 1
        return min(max(k, 0), bins - 1)

    px = [0]*bins
    py = [0]*bins
    pxy = [[0]*bins for _ in range(bins)]
    for x, y in zip(xs, ys):
        i = bin_index(x); j = bin_index(y)
        px[i] += 1; py[j] += 1; pxy[i][j] += 1
    px = [c / n for c in px]
    py = [c / n for c in py]
    pxy = [[c / n for c in row] for row in pxy]

    mi_nats = 0.0
    for i in range(bins):
        for j in range(bins):
            p = pxy[i][j]
            if p &gt; 0.0:
                mi_nats += p * math.log(p / ((px[i] + eps) * (py[j] + eps) + eps))
    mi_bits = mi_nats / math.log(2.0)
    return mi_nats, mi_bits

# ---------- Orchestrator ----------

    max_iter: int = 600,

    seed: int = 137,

    bins: int = 24,
    kappa: float = 0.02,            # tr√∂skel i nats
    collect_from_iter: int = 20
) -&gt; dict:
    if lambda_list is None:
        lambda_list = [0.25, 0.5, 1.0, 2.0, 4.0]

    M = Model(n=n, lam=lam_dynamics, seed=seed)
    I = [0.5 for _ in range(n)]
    history = [I[:]]
    samples: dict[float, dict[str, list[float]]] = {lam: {"x": [], "y": []} for lam in lambda_list}
    scaled_star = {lam: Scale(lam)(M.I_star) for lam in lambda_list}
    status = "EJ_KONVERGERAD"

    for t in range(1, max_iter + 1):
        I_next, info = M.step(I)
        dist = l2(I_next, I)
        history.append(I_next[:])

        if t &gt;= collect_from_iter:
            for lam in lambda_list:
                S = Scale(lam)
                I_N_scaled = S(info["I_N"])
                I_star_scaled = scaled_star[lam]
                samples[lam]["x"].extend(I_N_scaled)
                samples[lam]["y"].extend(I_star_scaled)

        I = I_next
        if dist &lt;= eps_stab and status != "STABIL":
            status = "STABIL"

    # MI per skala och minsta MI
    MI: dict[str, dict[str, float]] = {}
    min_mi = float("inf")
    for lam in lambda_list:
        mi_nats, mi_bits = mutual_information(samples[lam]["x"], samples[lam]["y"], bins=bins)
        MI[str(lam)] = {"nats": mi_nats, "bits": mi_bits, "num_samples": len(samples[lam]["x"])}
        if mi_nats &lt; min_mi:
            min_mi = mi_nats

    return {
        "status": status,
        "iterations": len(history) - 1,
        "epsilon_stability": eps_stab,
        "final_state": history[-1],
        "distance_last": l2(history[-1], history[-2]),
        "I_star": M.I_star,
        "lambda_list": lambda_list,

        "MI": MI,
        "kappa_nats": kappa,
        "min_MI_nats": min_mi,
        "S2_condition_passed": bool(min_mi &gt; kappa)
    }
